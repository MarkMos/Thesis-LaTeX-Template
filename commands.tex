\newcommand\note[2][]{%
	\if!#1!%
	\stepcounter{footnote}\footnotetext{#2}%
	\else%
	{\renewcommand\thefootnote{#1}%
		\footnotetext{#2}}%
	\fi}

%authors and affiliations
\newtoks\pauthtoks
\newcommand{\pauthor}[2][]{%
	\if!#1!%
	\pauthtoks=\expandafter{\the\pauthtoks#2\ }%
	\else
	\pauthtoks=\expandafter{\the\pauthtoks#2$^{#1}$\ }%
	\fi
}

\newtoks\paffiltoks\newif\ifpaffil\paffilfalse
\newcommand{\paffiliation}[2][]{%
	\paffiltrue
	\if!#1!%
	\paffiltoks=\expandafter{\the\paffiltoks{\item[]#2}}%
	\else
	\paffiltoks=\expandafter{\the\paffiltoks{\item[$^{#1}$]#2}}%
	\fi
}
